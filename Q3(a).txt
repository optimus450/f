face_value = 100  # Assumed standard face value
coupon = 8  # 8% of 100
y = 0.11  # 11% continuously compounded yield

t = np.array([1, 2, 3, 4, 5])

cash_flows = np.array([8, 8, 8, 8, 108])

Calculate the cost and provide profit tables (ignoring discounting) for:


(a) Bull spread (Calls: $25, $30; 6m) 

(b) Bear spread (Puts: $25, $30; 6m) 

(c & d) Butterfly spread (Calls/Puts: $25, $30, $35; 1y) 

(e) Straddle (Strike $30; 6m) 

(f) Strangle (Strikes $25, $35; 6m

import numpy as np
import pandas as pd
from scipy.stats import norm

# 1. Given parameters
S0 = 32       # Current stock price
sigma = 0.30  # Volatility (30%)
r = 0.05      # Risk-free rate (5%)

# 2. Black-Scholes Pricing Function
def bsm_price(K, T, is_call=True):
    d1 = (np.log(S0 / K) + (r + 0.5 * sigma**2) * T) / (sigma * np.sqrt(T))
    d2 = d1 - sigma * np.sqrt(T)
    
    if is_call:
        return S0 * norm.cdf(d1) - K * np.exp(-r * T) * norm.cdf(d2)
    else:
        return K * np.exp(-r * T) * norm.cdf(-d2) - S0 * norm.cdf(-d1)

# 3. Calculate necessary option premiums
# 6-month options (T=0.5)
c25_6m = bsm_price(25, 0.5)
c30_6m = bsm_price(30, 0.5)
c35_6m = bsm_price(35, 0.5)
p25_6m = bsm_price(25, 0.5, False)
p30_6m = bsm_price(30, 0.5, False)

# 1-year options (T=1.0)
c25_1y = bsm_price(25, 1.0)
c30_1y = bsm_price(30, 1.0)
c35_1y = bsm_price(35, 1.0)
p25_1y = bsm_price(25, 1.0, False)
p30_1y = bsm_price(30, 1.0, False)
p35_1y = bsm_price(35, 1.0, False)

# 4. Calculate Setup Costs for each strategy
cost_bull = c25_6m - c30_6m                     # (a) Buy 25 Call, Sell 30 Call
cost_bear = p30_6m - p25_6m                     # (b) Buy 30 Put, Sell 25 Put
cost_fly_c = c25_1y - 2*c30_1y + c35_1y         # (c) Buy 25, Sell 2x 30, Buy 35 (Calls)
cost_fly_p = p35_1y - 2*p30_1y + p25_1y         # (d) Buy 35, Sell 2x 30, Buy 25 (Puts)
cost_straddle = c30_6m + p30_6m                 # (e) Buy 30 Call, Buy 30 Put
cost_strangle = p25_6m + c35_6m                 # (f) Buy 25 Put, Buy 35 Call

print(f"--- Setup Costs ---")
print(f"(a) Bull Spread:  ${cost_bull:.2f}")
print(f"(b) Bear Spread:  ${cost_bear:.2f}")
print(f"(c) Butterfly (C):${cost_fly_c:.2f}")
print(f"(d) Butterfly (P):${cost_fly_p:.2f}")
print(f"(e) Straddle:     ${cost_straddle:.2f}")
print(f"(f) Strangle:     ${cost_strangle:.2f}\n")

# 5. Generate Profit Table
ST = np.array([20, 25, 30, 35, 40]) # Representative final stock prices

# Helper payoff functions
def call(K): return np.maximum(ST - K, 0)
def put(K): return np.maximum(K - ST, 0)

# Create DataFrame
df = pd.DataFrame({'ST': ST})
df['(a) Bull'] = (call(25) - call(30)) - cost_bull
df['(b) Bear'] = (put(30) - put(25)) - cost_bear
df['(c) Fly(C)'] = (call(25) - 2*call(30) + call(35)) - cost_fly_c

df['(e) Straddle'] = (call(30) + put(30)) - cost_straddle
df['(f) Strangle'] = (put(25) + call(35)) - cost_strangle

print("--- Profit Table ---")
print(df.round(2).to_string(index=False))

# 2. Feature Mapping up to 6th degree
def map_features(x1, x2, degree=6):
    m = x1.shape[0]
    output = np.ones((m, 1))
    for i in range(1, degree+1):
        for j in range(i+1):
            term = (x1**(i-j) * x2**j).reshape(m, 1)
            output = np.hstack((output, term))
    return output

# 3. Sigmoid
def sigmoid(z):
    return 1 / (1 + np.exp(-z))

# 4. Regularized Cost Function
def compute_cost_reg(X, y, theta, lambda_):
    m = len(y)
    h = sigmoid(X @ theta)
    reg_term = (lambda_ / (2 * m)) * np.sum(theta[1:] ** 2)
    cost = (-y @ np.log(h) - (1 - y) @ np.log(1 - h)) / m + reg_term
    return cost

# 5. Regularized Gradient Descent
def gradient_descent_reg(X, y, theta, alpha, num_iters, lambda_):
    m = len(y)
    cost_history = []

    for _ in range(num_iters):
        h = sigmoid(X @ theta)
        grad = (X.T @ (h-y)) / m
        grad[1:] += (lambda_ / m) * theta[1:]
        theta -= alpha * grad
        cost_history.append(compute_cost_reg(X, y, theta, lambda_))

    return theta, cost_history

# 6. Plot Decision Boundary

    u = np.linspace(-1, 1, 100)
    v = np.linspace(-1, 1, 100)

    for i in range(len(u)):
        for j in range(len(v)):
            mapped = map_features(np.array([u[i]]), np.array([v[j]]))
            z[i, j] = mapped @ theta

    pos = y == 1

digits = load_digits()
X = digits.data  # shape: (1797, 64) — 64 features (8x8 images)
y = digits.target  # labels: 0-9

# Sigmoid function
def sigmoid(z):
    return 1 / (1 + np.exp(-z))

# Cost and gradient for logistic regression
def cost_function(theta, X, y, reg_lambda):
    m = len(y)
    h = sigmoid(X @ theta)
    epsilon = 1e-5  # for numerical stability
    cost = (-y @ np.log(h + epsilon) - (1 - y) @ np.log(1 - h + epsilon)) / m
    reg = (reg_lambda / (2 * m)) * np.sum(theta[1:] ** 2)
    return cost + reg

def gradient(theta, X, y, reg_lambda):
    m = len(y)
    h = sigmoid(X @ theta)
    grad = (X.T @ (h - y)) / m
    grad[1:] += (reg_lambda / m) * theta[1:]
    return grad

# Add intercept term
X = np.insert(X, 0, 1, axis=1)  # shape: (1797, 65)

# One-vs-All Training
def one_vs_all(X, y, num_labels, reg_lambda=0.1):
    m, n = X.shape
    all_theta = np.zeros((num_labels, n))
    
    from scipy.optimize import minimize
    for c in range(num_labels):
        y_c = (y == c).astype(int)  # binary for class c
        theta0 = np.zeros(n)
        res = minimize(fun=cost_function, x0=theta0, args=(X, y_c, reg_lambda), 
                       jac=gradient, method='TNC', options={'maxiter': 300})
        all_theta[c] = res.x
    return all_theta

# Prediction
def predict_one_vs_all(all_theta, X):
    probs = sigmoid(X @ all_theta.T)  # shape: (m, num_labels)
    return np.argmax(probs, axis=1)

# Train the OvA logistic regression
num_labels = 10  # digits 0-9
reg_lambda = 0.1
all_theta = one_vs_all(X, y, num_labels, reg_lambda)

# Predict and compute accuracy
y_pred = predict_one_vs_all(all_theta, X)
accuracy = accuracy_score(y, y_pred)

print("Training Accuracy (OvA):", accuracy * 100, "%")

# Display learned theta for each class (first 10 coefficients)
for i in range(num_labels):
    print(f"Class {i} θ (first 10):", all_theta[i][:10])
